namespace = autocolcul_events

province_event = {
	id = autocolcul_events.1
	title = autocolcul_events.1.t
	desc = autocolcul_events.1.d
    picture = none
	hidden = yes

	trigger = {
        has_owner_culture = no
        is_core = owner
        NOT = { nationalism = 1 }

		owner = {
			OR = {
				is_former_colonial_nation = yes
				is_colonial_nation = yes
			}
		}

		OR = {
			development = 20
			has_port = yes
			any_neighbor_province = {
				has_owner_culture = yes
				owned_by = ROOT
			}
		}

        AND = {
            culture = greek
            owner = { primary_culture = roman }
        }

		OR = {
			culture_group = andean_group
			culture_group = apachean
			culture_group = araucanian
			culture_group = aridoamerican
			culture_group = carribean
			culture_group = caddoan
			culture_group = central_algonquian
			culture_group = central_american
			culture_group = chibchan
			culture_group = eastern_algonquian
			culture_group = eskaleut
			culture_group = iroquoian
			culture_group = maranon
			culture_group = mataco
			culture_group = maya
			culture_group = muskogean
			culture_group = na_dene
			culture_group = otomanguean
			culture_group = penutian
			culture_group = plains_algonquian
			culture_group = siouan
			culture_group = sonoran
			culture_group = je_tupi
			culture_group = je
			culture_group = aboriginal_australian
			culture_group = pacific
		}
	}

	is_mtth_scaled_to_size = 1

	mean_time_to_happen = {
		months = 32

		modifier = {
			factor = 0.625
			owner = { stability = 3 }
		}
		modifier = {
			factor = 0.75
			owner = {
				stability = 2
				NOT = { stability = 3 }
			}
		}
		modifier = {
			factor = 0.875
			owner = {
				stability = 1
				NOT = { stability = 2 }
			}
		}
		modifier = {
			factor = 1.5
			owner = {
				NOT = { stability = 0 }
			}
		}
		modifier = {
			factor = 0.75
			is_year = 1700
		}
	}

	option = {
		name = autocolcul_events.1.a
		ai_chance = { factor = 5 }

		if = {
			limit = {
				culture = greek
				owner = { primary_culture = roman }
			}
			change_culture = roman
		}
		else = {
			change_culture = owner
		}
	}
}
